async function loadFFmpeg() { // if already created and loaded, nothing to do if (ffmpeg && ffmpeg.loaded) return; // ensure runtime globals exist if (!window.FFmpegWASM) { throw new Error("FFmpeg failed to load. Please make sure ffmpeg.js and util.js are uploaded to your GitHub repository."); } const { FFmpeg } = window.FFmpegWASM; log('Loading FFmpeg engine (core files)...', 'log-info'); // try load from local files first (repo), fallback to CDN if needed const tries = [ { coreURL: './ffmpeg-core.js', wasmURL: './ffmpeg-core.wasm', label: 'local' }, { coreURL: 'https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.js', wasmURL: 'https://unpkg.com/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.wasm', label: 'unpkg' }, { coreURL: 'https://cdn.jsdelivr.net/npm/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.js', wasmURL: 'https://cdn.jsdelivr.net/npm/@ffmpeg/core@0.12.6/dist/umd/ffmpeg-core.wasm', label: 'jsdelivr' } ]; let lastErr = null; for (const t of tries) { try { log(`Attempting to load core from: ${t.label}`, 'text-gray-500'); // Create a FRESH instance for every attempt ffmpeg = new FFmpeg(); // Attach event handlers to the new instance ffmpeg.on('log', (m) => { const message = (m && m.message) ? m.message : (typeof m === 'string' ? m : JSON.stringify(m)); if (message.includes('bitrate') || message.includes('frame=') || message.includes('size=')) { log(message, 'text-gray-400 text-[11px]'); } }); ffmpeg.on('progress', (p) => { const prog = (p && (p.progress ?? p)) || 0; const pct = Math.max(0, Math.min(100, Math.round(prog * 100))); progressBar.style.width = `${pct}%`; progressText.textContent = `${pct}%`; }); // Attempt to load await ffmpeg.load({ coreURL: t.coreURL, wasmURL: t.wasmURL }); log(`FFmpeg engine loaded successfully from ${t.label}!`, 'log-success'); return; // Exit the loop on success } catch (err) { lastErr = err; log(`Failed to load core from ${t.label}: ${err.message}`, 'log-error'); // The loop will automatically continue to the next attempt } } // If the loop finishes and nothing worked: throw new Error(`FFmpeg failed to load from all locations. Last error: ${lastErr?.message || 'unknown'}`); }
